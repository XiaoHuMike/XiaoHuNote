# Bartender不独占进程

# 1. 场景需求

客户包装会先开着bartender软件打印特殊内包标签，接着再用我们系统功能打印外包标签，这时bartender软件就会被系统关闭,影响用户体验

# 2. 问题分析

查看SMOM源码发现Bartender打印是使用process+命令行参数以调用bartender进行打印,打印完成后会自动关闭bartender进程。

# 3. 解决思路

利用Bartender类库方式进行打印,此方式不会影响用户本身对Bartender软件的操作

# 4. 解决步骤

# 1. 引入类库 Seagull.BarTender.Print.Dll

ps：此库的版本必须与电脑安装的bartender版本一致，可在软件安装目录下获取。

# 2. 代码

public void BartenderPrint(IEnumerable<object> datas, IReport report, IPrintable printable, string filePath, string printJobName, string printer, short copies, Action action)
{
    //生成data.csv文件,btw模板会设置数据源来自这个csv文件路径
    report.GenerateDataFile(printable, datas);
    //创建Bartender应用程序对象
    Engine btEngine = new Engine();
    string printerName = printer;
    string tmplFileName = filePath;
    //启用Bartender应用程序
    if (!btEngine.IsAlive)
        btEngine.Restart(SaveOptions.DoNotSaveChanges);
    //打开btw文件
    LabelFormatDocument btFormat = btEngine.Documents.Open(tmplFileName);
    //传入打印机名称
    btFormat.PrintSetup.PrinterName = printerName;
    //打印张数
    btFormat.PrintSetup.IdenticalCopiesOfLabel = copies;
    //打印
    Messages messages;
    Result printResult = btFormat.Print(printJobName, out messages);
    //关闭标签
    btFormat.Close(SaveOptions.DoNotSaveChanges);
    //按需关闭Bartender应用程序
    btEngine.Dispose();
    //判断打印是否成功
    if (printResult != Result.Success)
    {
        string errorMsg = string.Empty;
        for (int i = 0; i < messages.Count; i++)
        {
            errorMsg += messages[i].Text + Environment.NewLine;
        }
        throw new ValidationException(errorMsg);
    }
    //打印成功后回调
    action();
}
